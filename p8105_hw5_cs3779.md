replace_missing = function(x) {
  if (is.numeric(x)) {
    x = round(replace_na(x, mean(x, na.rm = TRUE)), digits = 1)
  }
  else if (is.character(x)) {
    (x = replace_na(x, "virginica"))
  }
}
